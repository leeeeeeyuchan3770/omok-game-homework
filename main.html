<script>

// ===== 연결리스트 =====

class Node {

    constructor(x,y,player){

        this.x = x;

        this.y = y;

        this.player = player;

        this.next = null;

    }

}


class LinkedList {

    constructor(){

        this.head = null;

        this.tail = null;

    }


    append(x,y,player){

        const n = new Node(x,y,player);

        if(!this.head){

            this.head = this.tail = n;

        } else {

            this.tail.next = n;

            this.tail = n;

        }

    }


    toArray(){

        let arr = [];

        let cur = this.head;

        while(cur){

            arr.push(cur);

            cur = cur.next;

        }

        return arr;

    }

}


// ===== 게임 설정 =====

const SIZE = 15;

const CELL = 40;


let board = Array.from({length: SIZE}, () => Array(SIZE).fill(0));

let turn = 1;


const moves = new LinkedList();


const canvas = document.getElementById("board");

const ctx = canvas.getContext("2d");


// ===== 보드 그리기 =====

function drawBoard(){

    for(let i=0;i<SIZE;i++){

        ctx.beginPath();

        ctx.moveTo(CELL/2, CELL/2 + i*CELL);

        ctx.lineTo(CELL/2+(SIZE-1)*CELL, CELL/2+i*CELL);

        ctx.stroke();


        ctx.beginPath();

        ctx.moveTo(CELL/2+i*CELL, CELL/2);

        ctx.lineTo(CELL/2+i*CELL, CELL/2+(SIZE-1)*CELL);

        ctx.stroke();

    }

}


// ===== 돌 그리기 =====

function drawStone(x,y,player){

    ctx.beginPath();

    ctx.arc(CELL/2+x*CELL, CELL/2+y*CELL, 15, 0, Math.PI*2);

    ctx.fillStyle = player===1 ? "black" : "white";

    ctx.fill();

    ctx.stroke();

}


// ===== 승리 판정 =====

function checkWin(x,y){

    const dirs = [[1,0],[0,1],[1,1],[1,-1]];


    for(let [dx,dy] of dirs){

        let count = 1;


        for(let i=1;i<5;i++){

            let nx = x+dx*i, ny = y+dy*i;

            if(board[ny]?.[nx] === turn) count++;

            else break;

        }


        for(let i=1;i<5;i++){

            let nx = x-dx*i, ny = y-dy*i;

            if(board[ny]?.[nx] === turn) count++;

            else break;

        }


        if(count >= 5) return true;

    }

    return false;

}


// ===== 클릭 이벤트 =====

canvas.addEventListener("click",(e)=>{

    const rect = canvas.getBoundingClientRect();

    const x = Math.round((e.clientX-rect.left-CELL/2)/CELL);

    const y = Math.round((e.clientY-rect.top-CELL/2)/CELL);


    if(board[y][x] !== 0) return;


    board[y][x] = turn;

    moves.append(x,y,turn); // 🔥 연결리스트 사용

    drawStone(x,y,turn);

    renderMoves();


    if(checkWin(x,y)){

        setTimeout(()=>{

            alert((turn===1?"흑":"백")+" 승리!");

            location.reload();

        },100);

    }


    turn = turn===1 ? 2 : 1;

});


// ===== 수순 표시 =====

function renderMoves(){

    const listDiv = document.getElementById("list");

    listDiv.innerHTML = "";


    let arr = moves.toArray();


    arr.forEach((m)=>{

        let d = document.createElement("div");

        d.className = "move";

        d.innerText = (m.player===1?"●":"○") + " ("+m.x+","+m.y+")";

        listDiv.appendChild(d);

    });

}


// ===== 실행 =====

drawBoard();
